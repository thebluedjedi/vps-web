// Chart configurations const chartDefaults = { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: false } }, scales: { x: { display: false, grid: { display: false } }, y: { display: false, grid: { display: false } } }, elements: { point: { radius: 0, hoverRadius: 3 }, line: { tension: 0.4, borderWidth: 2 } } }; // Generate time labels const timeLabels = Array.from({length: 20}, (_, i) => { const time = new Date(Date.now() - (19 - i) * 30000); return time.toLocaleTimeString(); }); // Initialize charts const cpuChart = new Chart(document.getElementById('cpuChart'), { type: 'line', data: { labels: timeLabels, datasets: [{ data: [], borderColor: '#4fc3f7', backgroundColor: 'rgba(79, 195, 247, 0.1)', fill: true }] }, options: chartDefaults }); const memChart = new Chart(document.getElementById('memChart'), { type: 'line', data: { labels: timeLabels, datasets: [{ data: [], borderColor: '#29b6f6', backgroundColor: 'rgba(41, 182, 246, 0.1)', fill: true }] }, options: chartDefaults }); const storageChart = new Chart(document.getElementById('storageChart'), { type: 'doughnut', data: { datasets: [{ data: [0, 100], backgroundColor: ['#4fc3f7', 'rgba(255, 255, 255, 0.1)'], borderWidth: 0 }] }, options: { responsive: true, maintainAspectRatio: false, cutout: '75%', plugins: { legend: { display: false }, tooltip: { enabled: false } } } }); const networkChart = new Chart(document.getElementById('networkChart'), { type: 'line', data: { labels: timeLabels, datasets: [{ data: [], borderColor: '#81c784', backgroundColor: 'rgba(129, 199, 132, 0.1)', fill: true }] }, options: chartDefaults }); const responseChart = new Chart(document.getElementById('responseChart'), { type: 'line', data: { labels: timeLabels, datasets: [{ data: [], borderColor: '#ff9800', backgroundColor: 'rgba(255, 152, 0, 0.1)', fill: true }] }, options: chartDefaults }); // Tunnel management function checkTunnelStatus(service, port) { const statusDot = document.getElementById(`${service}-status`); fetch(`http://localhost:${port}/`, { mode: 'no-cors', cache: 'no-cache' }) .then(() => { statusDot?.classList.add('available'); }) .catch(() => { statusDot?.classList.remove('available'); }); } function openTunnel(service, url, tunnelCmd) { // Try to open the URL window.open(url, '_blank'); // Check if tunnel is running setTimeout(() => { fetch(url, { mode: 'no-cors', cache: 'no-cache' }) .catch(() => { alert(`SSH tunnel required for ${service}:\n\nssh -L ${tunnelCmd} -p 2222 root@161.97.82.115\n\nThen visit: ${url}`); }); }, 100); } // Check tunnel status on load window.addEventListener('load', () => { checkTunnelStatus('traefik', 8080); checkTunnelStatus('grafana', 3000); checkTunnelStatus('portainer', 9000); // Recheck every 10 seconds setInterval(() => { checkTunnelStatus('traefik', 8080); checkTunnelStatus('grafana', 3000); checkTunnelStatus('portainer', 9000); }, 10000); }); // Prometheus queries async function queryPrometheus(query) { try { const url = `/api/prometheus/api/v1/query?query=${encodeURIComponent(query)}`; const response = await fetch(url); if (!response.ok) { throw new Error(`HTTP ${response.status}`); } const data = await response.json(); if (data.status !== 'success') { throw new Error('Query failed'); } return data.data; } catch (error) { console.error(`Prometheus query failed: ${query}`, error); return null; } } async function queryPrometheusRange(query) { try { const endTime = Math.floor(Date.now() / 1000); const url = `/api/prometheus/api/v1/query_range?query=${encodeURIComponent(query)}&start=${endTime - 600}&end=${endTime}&step=30`; const response = await fetch(url); if (!response.ok) { throw new Error(`HTTP ${response.status}`); } const data = await response.json(); if (data.status !== 'success') { throw new Error('Range query failed'); } return data.data; } catch (error) { console.error(`Prometheus range query failed: ${query}`, error); return null; } } function updateChartData(chart, values) { if (values && values.length > 0) { chart.data.datasets[0].data = values.slice(-20); chart.update('none'); } } // Check service status async function checkServices() { // Query node exporter up status as a proxy for container health const uptimeData = await queryPrometheus('up'); // Update service lights based on container status Object.keys(serviceContainers).forEach(service => { const light = document.getElementById(`${service}-light`); if (light) { // For now, assume all are online if we get prometheus data // You could enhance this with specific container checks light.classList.add('online'); light.classList.remove('offline'); } }); } // Simplified uptime function async function getUptime() { const uptimeData = await queryPrometheus('(time() - process_start_time_seconds{job="node-exporter"})'); if (uptimeData && uptimeData.result && uptimeData.result[0]) { const seconds = parseFloat(uptimeData.result[0].value[1]); const days = Math.floor(seconds / 86400); const hours = Math.floor((seconds % 86400) / 3600); const minutes = Math.floor((seconds % 3600) / 60); const fullUptime = `${days}d ${hours}h ${minutes}m`; const shortUptime = days > 0 ? `${days}d ${hours}h` : `${hours}h ${minutes}m`; document.getElementById('uptimeValue').textContent = fullUptime; document.getElementById('uptimeDisplay').textContent = shortUptime; } } // Refresh all metrics async function refreshMetrics() { // Get uptime await getUptime(); // CPU Usage const cpuData = await queryPrometheusRange('100 - (avg(rate(node_cpu_seconds_total{mode="idle"}[5m])) * 100)'); if (cpuData && cpuData.result && cpuData.result[0] && cpuData.result[0].values) { const cpuValues = cpuData.result[0].values.map(v => parseFloat(v[1])); updateChartData(cpuChart, cpuValues); const latestCpu = cpuValues[cpuValues.length - 1]; document.getElementById('cpuValue').textContent = latestCpu.toFixed(1) + '%'; } // Memory Usage const memData = await queryPrometheusRange('(1 - (node_memory_MemAvailable_bytes / node_memory_MemTotal_bytes)) * 100'); if (memData && memData.result && memData.result[0] && memData.result[0].values) { const memValues = memData.result[0].values.map(v => parseFloat(v[1])); updateChartData(memChart, memValues); const latestMem = memValues[memValues.length - 1]; document.getElementById('memValue').textContent = latestMem.toFixed(1) + '%'; } // Storage Usage const storageData = await queryPrometheus('(1 - (node_filesystem_avail_bytes{fstype!="tmpfs"} / node_filesystem_size_bytes{fstype!="tmpfs"})) * 100'); if (storageData && storageData.result && storageData.result[0]) { const storageValue = parseFloat(storageData.result[0].value[1]); storageChart.data.datasets[0].data = [storageValue, 100 - storageValue]; storageChart.update('none'); document.getElementById('storageValue').textContent = storageValue.toFixed(1) + '%'; } // Network Traffic const networkData = await queryPrometheusRange('rate(node_network_receive_bytes_total{device!="lo"}[5m]) + rate(node_network_transmit_bytes_total{device!="lo"}[5m])'); if (networkData && networkData.result && networkData.result[0] && networkData.result[0].values) { const networkValues = networkData.result[0].values.map(v => parseFloat(v[1]) / 1024 / 1024); updateChartData(networkChart, networkValues); const latestNetwork = networkValues[networkValues.length - 1]; document.getElementById('networkValue').textContent = latestNetwork.toFixed(2) + ' MB/s'; } // Response Times const responseData = await queryPrometheusRange('histogram_quantile(0.95, rate(prometheus_http_request_duration_seconds_bucket[5m]))'); if (responseData && responseData.result && responseData.result[0] && responseData.result[0].values) { const responseValues = responseData.result[0].values.map(v => parseFloat(v[1]) * 1000); updateChartData(responseChart, responseValues); const latestResponse = responseValues[responseValues.length - 1]; document.getElementById('responseValue').textContent = Math.round(latestResponse) + 'ms'; } } // Refresh all with loading indicator async function refreshAll() { const btn = document.getElementById('refreshBtn'); const icon = document.getElementById('refreshIcon'); btn.disabled = true; icon.classList.add('spinning'); await Promise.all([ refreshMetrics(), checkServices() ]); btn.disabled = false; icon.classList.remove('spinning'); } // Auto refresh every 30 seconds setInterval(refreshAll, 30000); // Initial load document.addEventListener('DOMContentLoaded', function() { console.log('Blue Djedi Admin Dashboard loaded'); refreshAll(); });
